for i in input/individual_fasta/*.fasta; do
    variants=$(basename "$i" .fasta)   
    blastn -query $i -db cpe_blast_db/cpe_database -outfmt '10 qseqid qlen sseqid slen pident mismatch gaps qstart qend length sstart send evalue' -out "results/${variants}_blast.tsv"
done
